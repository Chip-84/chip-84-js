: main
	hires
  i := splash
  draw-bitmap

	v2 := 0
  loop
		scroll-up 2
		v2 += 1
		
		if v2 == 6 then finish-scrolling
		if v2 == 30 then jump end-animation

    wait-for-frame
  again
	
: end-animation
	loop
		v0 := random 127
		v1 := 0
		i := pixel
		sprite v0 v1 1
		scroll-down 1
		
		wait-for-frame
	again
	
: finish-scrolling
	v4 := 0x04
	v5 := 0x05
	v6 := 0x06
	v2 := 7
;

: wait-for-frame
	loop
		vf := delay
		if vf != 0 then
	again
	vf := 3
	delay := vf
;
	
: draw-bitmap
	clear
	v0 := 0 # x
	v1 := 14 # y
	v2 := 0 # byte
	v3 := 1 # constant
	loop
		sprite v0 v1 1
		i  += v3
		v2 += 1
		v0 += 8
		if v0 == 128 then v1 += 1
		if v0 == 128 then v0 := 0
		if v1 != 64 then
	again
;
